TASKS = {}
while True:
    print("TO DO LIST APPLICATION:")
    print("1. Add a new task")
    print("2. Remove a task")
    print("3. Mark a task as complete")
    print("4. View all tasks")
    print("5. Exit")
    choice = int(input("Choose any one option: "))
    if choice == 1:
        new = input("Enter the name of the task: ")
        TASKS[new] = "not completed"
        print("New task added")
    elif choice == 2:
        remove = input("Enter the name of the task to be removed: ")
        if remove in TASKS:
            TASKS.pop(remove)
            print("The Specified Task is Removed!")
        else:
            print("Oops, task is not found")
    elif choice == 3:
        completed = input("Enter the task to mark as completed: ")
        if completed in TASKS:
            TASKS[completed] = "completed"
            print("Task marked as completed")
        else:
            print("Oops, task is not found")
    elif choice == 4:
        print(TASKS)
    elif choice == 5:
        break
    else:
        print("Invalid choice")
